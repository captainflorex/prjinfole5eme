116 /* the following bit-flags are computed once by the glyph */
117 /* analyzer, for both dimensions */
118 enum
119 {
120 PSH3_POINT_OFF = 1, /* point is off the curve */
121 PSH3_POINT_SMOOTH = 2, /* point is smooth */
122 PSH3_POINT_INFLEX = 4 /* point is inflection */
123 };
124
125#define psh3_point_is_smooth( p ) ( (p)->flags & PSH3_POINT_SMOOTH )
126#define psh3_point_is_off( p ) ( (p)->flags & PSH3_POINT_OFF )
127#define psh3_point_is_inflex( p ) ( (p)->flags & PSH3_POINT_INFLEX )
128
129#define psh3_point_set_smooth( p ) (p)->flags |= PSH3_POINT_SMOOTH
130#define psh3_point_set_off( p ) (p)->flags |= PSH3_POINT_OFF
131#define psh3_point_set_inflex( p ) (p)->flags |= PSH3_POINT_INFLEX
132
133 /* the following bit-flags are re-computed for each dimension */
134 enum
135 {
136 PSH3_POINT_STRONG = 16, /* point is strong */
137 PSH3_POINT_FITTED = 32, /* point is already fitted */
138 PSH3_POINT_EXTREMUM = 64, /* point is local extremum */
139 PSH3_POINT_POSITIVE = 128, /* extremum has positive contour flow */
140 PSH3_POINT_NEGATIVE = 256, /* extremum has negative contour flow */
141 PSH3_POINT_EDGE_MIN = 512, /* point is aligned to left/bottom stem edge */
142 PSH3_POINT_EDGE_MAX = 1024 /* point is aligned to top/right stem edge */
143 };
144
145#define psh3_point_is_strong( p ) ( (p)->flags2 & PSH3_POINT_STRONG )
146#define psh3_point_is_fitted( p ) ( (p)->flags2 & PSH3_POINT_FITTED )
147#define psh3_point_is_extremum( p ) ( (p)->flags2 & PSH3_POINT_EXTREMUM )
148#define psh3_point_is_positive( p ) ( (p)->flags2 & PSH3_POINT_POSITIVE )
149#define psh3_point_is_negative( p ) ( (p)->flags2 & PSH3_POINT_NEGATIVE )
150#define psh3_point_is_edge_min( p ) ( (p)->flags2 & PSH3_POINT_EDGE_MIN )
151#define psh3_point_is_edge_max( p ) ( (p)->flags2 & PSH3_POINT_EDGE_MAX )
152
153#define psh3_point_set_strong( p ) (p)->flags2 |= PSH3_POINT_STRONG
154#define psh3_point_set_fitted( p ) (p)->flags2 |= PSH3_POINT_FITTED
155#define psh3_point_set_extremum( p ) (p)->flags2 |= PSH3_POINT_EXTREMUM
156#define psh3_point_set_positive( p ) (p)->flags2 |= PSH3_POINT_POSITIVE

79 FT_CALLBACK_DEF( FT_UInt )
80 bdf_cmap_char_index( BDF_CMap cmap,
81 FT_UInt32 charcode )
82 {
83 BDF_encoding_el* encodings = cmap->encodings;
84 FT_UInt min, max, mid;
85 FT_UInt result = 0;
86
87
88 min = 0;
89 max = cmap->num_encodings;
90
91 while ( min < max )
92 {
93 FT_UInt32 code;
94
95
96 mid = ( min + max ) >> 1;
97 code = encodings[mid].enc;
98
99 if ( charcode == code )
100 {
101 result = encodings[mid].glyph + 1;
102 break;
103 }
104
105 if ( charcode < code )
106 max = mid;
107 else
108 min = mid + 1;
109 }
110
111 return result;
112 }

117 FT_UInt32 *acharcode )
118 {
119 BDF_encoding_el* encodings = cmap->encodings;
120 FT_UInt min, max, mid;
121 FT_UInt32 charcode = *acharcode + 1;
122 FT_UInt result = 0;
123
124
125 min = 0;
126 max = cmap->num_encodings;
127
128 while ( min < max )
129 {
130 FT_UInt32 code;
131
132
133 mid = ( min + max ) >> 1;
134 code = encodings[mid].enc;
135
136 if ( charcode == code )
137 {
138 result = encodings[mid].glyph + 1;
139 goto Exit;
140 }
141
142 if ( charcode < code )
143 max = mid;
144 else
145 min = mid + 1;
146 }
147
148 charcode = 0;
149 if ( min < cmap->num_encodings )
150 {
151 charcode = encodings[min].enc;
152 result = encodings[min].glyph + 1;
153 }
154
155 Exit:
156 *acharcode = charcode;
157 return result;
158 }

197 FT_CALLBACK_DEF( FT_Error )
198 BDF_Face_Init( FT_Stream stream,
199 BDF_Face face,
200 FT_Int face_index,
201 FT_Int num_params,
202 FT_Parameter* params )
203 {
204 FT_Error error = BDF_Err_Ok;
205 FT_Memory memory = FT_FACE_MEMORY( face );
206
207 bdf_font_t* font;
208 bdf_options_t options;
209
210 FT_UNUSED( num_params );
211 FT_UNUSED( params );
212 FT_UNUSED( face_index );
213
214
215 if ( FT_STREAM_SEEK( 0 ) )
216 goto Exit;
217
218 options.correct_metrics = 1; /* FZ XXX: options semantics */
219 options.keep_unencoded = 1;
220 options.keep_comments = 0;
221 options.font_spacing = BDF_PROPORTIONAL;
222
223 error = bdf_load_font( stream, memory, &options, &font );
224 if ( error == BDF_Err_Missing_Startfont_Field )
225 {
226 FT_TRACE2(( "[not a valid BDF file]\n" ));
227 goto Fail;
228 }
229 else if ( error )
230 goto Exit;
231
232 /* we have a bdf font: let's construct the face object */
233 face->bdffont = font;
234 {
235 FT_Face root = FT_FACE( face );
236 bdf_property_t* prop = NULL;
237
238
239 FT_TRACE4(( "number of glyphs: %d (%d)\n",
240 font->glyphs_size,
241 font->glyphs_used ));
242 FT_TRACE4(( "number of unencoded glyphs: %d (%d)\n",
243 font->unencoded_size,
244 font->unencoded_used ));
245
246 root->num_faces = 1;
247 root->face_index = 0;
248 root->face_flags = FT_FACE_FLAG_FIXED_SIZES |
249 FT_FACE_FLAG_HORIZONTAL |
250 FT_FACE_FLAG_FAST_GLYPHS;
251
252 prop = bdf_get_font_property( font, (char *)"SPACING" );
253 if ( prop != NULL )
254 if ( prop->format == BDF_ATOM )
255 if ( prop->value.atom != NULL )
256 if ( ( *(prop->value.atom) == 'M' ) ||
257 ( *(prop->value.atom) == 'C' ) )
258 root->face_flags |= FT_FACE_FLAG_FIXED_WIDTH;
259
260 /* FZ XXX: TO DO: FT_FACE_FLAGS_VERTICAL */
261 /* FZ XXX: I need a font to implement this */
262
263 root->style_flags = 0;
264 prop = bdf_get_font_property( font, (char *)"SLANT" );
265 if ( prop != NULL )
266 if ( prop->format == BDF_ATOM )
267 if ( prop->value.atom != NULL )
268 if ( ( *(prop->value.atom) == 'O' ) ||
269 ( *(prop->value.atom) == 'I' ) )
270 root->style_flags |= FT_STYLE_FLAG_ITALIC;
271
272 prop = bdf_get_font_property( font, (char *)"WEIGHT_NAME" );
273 if ( prop != NULL )
274 if ( prop->format == BDF_ATOM )
275 if ( prop->value.atom != NULL )
276 if ( *(prop->value.atom) == 'B' )
277 root->style_flags |= FT_STYLE_FLAG_BOLD;
278
279 prop = bdf_get_font_property( font, (char *)"FAMILY_NAME" );
280 if ( ( prop != NULL ) && ( prop->value.atom != NULL ) )
281 {
282 int l = ft_strlen( prop->value.atom ) + 1;
283
284
285 if ( FT_NEW_ARRAY( root->family_name, l ) )
286 goto Exit;
287 ft_strcpy( root->family_name, prop->value.atom );
288 }
289 else
290 root->family_name = 0;
291
292 root->style_name = (char *)"Regular";
293 if ( root->style_flags & FT_STYLE_FLAG_BOLD )
294 {
295 if ( root->style_flags & FT_STYLE_FLAG_ITALIC )
296 root->style_name = (char *)"Bold Italic";
297 else
298 root->style_name = (char *)"Bold";
299 }
300 else if ( root->style_flags & FT_STYLE_FLAG_ITALIC )
301 root->style_name = (char *)"Italic";
302
303 root->num_glyphs = font->glyphs_size; /* unencoded included */
304
305 root->num_fixed_sizes = 1;
306 if ( FT_NEW_ARRAY( root->available_sizes, 1 ) )
307 goto Exit;
308
309 prop = bdf_get_font_property( font, (char *)"AVERAGE_WIDTH" );
310 if ( ( prop != NULL ) && ( prop->value.int32 >= 10 ) )
311 root->available_sizes->width = (short)( prop->value.int32 / 10 );
312
313 prop = bdf_get_font_property( font, (char *)"PIXEL_SIZE" );
314 if ( prop != NULL )
315 root->available_sizes->height = (short) prop->value.int32;
316 else
317 {
318 prop = bdf_get_font_property( font, (char *)"POINT_SIZE" );
319 if ( prop != NULL )
320 {
321 bdf_property_t *yres;
322
323
324 yres = bdf_get_font_property( font, (char *)"RESOLUTION_Y" );
325 if ( yres != NULL )
326 {
327 FT_TRACE4(( "POINT_SIZE: %d RESOLUTION_Y: %d\n",
328 prop->value.int32, yres->value.int32 ));
329 root->available_sizes->height =
330 (FT_Short)( prop->value.int32 * yres->value.int32 / 720 );
331 }
332 }
333 }
334
335 if ( root->available_sizes->width == 0 )
336 {
337 if ( root->available_sizes->height == 0 )
338 {
339 /* some fonts have broken SIZE declaration (jiskan24.bdf) */
340 FT_ERROR(( "BDF_Face_Init: reading size\n" ));
341 root->available_sizes->width = (FT_Short)font->point_size;
342 }
343 else
344 root->available_sizes->width = root->available_sizes->height;
345 }
346 if ( root->available_sizes->height == 0 )
347 root->available_sizes->height = root->available_sizes->width;
348
349 /* encoding table */
350 {
351 bdf_glyph_t* cur = font->glyphs;
352 unsigned long n;
353
354
355 if ( FT_NEW_ARRAY( face->en_table, font->glyphs_size ) )
356 goto Exit;
357
358 for ( n = 0; n < font->glyphs_size; n++ )
359 {
360 (face->en_table[n]).enc = cur[n].encoding;
361 FT_TRACE4(( "idx %d, val 0x%lX\n", n, cur[n].encoding ));
362 (face->en_table[n]).glyph = (FT_Short)n;
363 }
364 }
365
366 /* charmaps */
367 {
368 bdf_property_t *charset_registry = 0, *charset_encoding = 0;
369 FT_Bool unicode_charmap = 0;
370
371
372 charset_registry =
373 bdf_get_font_property( font, (char *)"CHARSET_REGISTRY" );
374 charset_encoding =
375 bdf_get_font_property( font, (char *)"CHARSET_ENCODING" );
376 if ( ( charset_registry != NULL ) && ( charset_encoding != NULL ) )
377 {
378 if ( ( charset_registry->format == BDF_ATOM ) &&
379 ( charset_encoding->format == BDF_ATOM ) &&
380 ( charset_registry->value.atom != NULL ) &&
381 ( charset_encoding->value.atom != NULL ) )
382 {
383 if ( FT_NEW_ARRAY( face->charset_encoding,
384 strlen( charset_encoding->value.atom ) + 1 ) )
385 goto Exit;
386 if ( FT_NEW_ARRAY( face->charset_registry,
387 strlen( charset_registry->value.atom ) + 1 ) )
388 goto Exit;
389 ft_strcpy( face->charset_registry, charset_registry->value.atom );
390 ft_strcpy( face->charset_encoding, charset_encoding->value.atom );
391 if ( !ft_strcmp( face->charset_registry, "ISO10646" ) ||
392 ( !ft_strcmp( face->charset_registry, "ISO8859" ) &&
393 !ft_strcmp( face->charset_encoding, "1" ) ) )
394 unicode_charmap = 1;
395
396 {
397 FT_CharMapRec charmap;
398
399
400 charmap.face = FT_FACE( face );
401 charmap.encoding = FT_ENCODING_NONE;
402 charmap.platform_id = 0;
403 charmap.encoding_id = 0;
404
405 if ( unicode_charmap )
406 {
407 charmap.encoding = FT_ENCODING_UNICODE;
408 charmap.platform_id = 3;
409 charmap.encoding_id = 1;
410 }
411
412 error = FT_CMap_New( &bdf_cmap_class, NULL, &charmap, NULL );
413
414#if 0
415 /* Select default charmap */
416 if (root->num_charmaps)
417 root->charmap = root->charmaps[0];
418#endif
419 }
420
421 goto Exit;
422 }
423 }
424
425 /* otherwise assume Adobe standard encoding */
426
427 {
428 FT_CharMapRec charmap;
429
430
431 charmap.face = FT_FACE( face );
432 charmap.encoding = FT_ENCODING_ADOBE_STANDARD;
433 charmap.platform_id = 7;
434 charmap.encoding_id = 0;
435
436 error = FT_CMap_New( &bdf_cmap_class, NULL, &charmap, NULL );
437
438 /* Select default charmap */
439 if (root->num_charmaps)
440 root->charmap = root->charmaps[0];
441 }
442 }
443 }
444
445 Exit:
446 return error;
447
448 Fail:
449 BDF_Face_Done( face );
450 return BDF_Err_Unknown_File_Format;
451 }

454 static FT_Error
455 BDF_Set_Pixel_Size( FT_Size size )
456 {
457 BDF_Face face = (BDF_Face)FT_SIZE_FACE( size );
458 FT_Face root = FT_FACE( face );
459
460
461 FT_TRACE4(( "rec %d - pres %d\n",
462 size->metrics.y_ppem, root->available_sizes->height ));
463
464 if ( size->metrics.y_ppem == root->available_sizes->height )
465 {
466 size->metrics.ascender = face->bdffont->bbx.ascent << 6;
467 size->metrics.descender = face->bdffont->bbx.descent * ( -64 );
468 size->metrics.height = face->bdffont->bbx.height << 6;
469
470 return BDF_Err_Ok;
471 }
472 else
473 return BDF_Err_Invalid_Pixel_Size;
474 }

477 static FT_Error
478 BDF_Glyph_Load( FT_GlyphSlot slot,
479 FT_Size size,
480 FT_UInt glyph_index,
481 FT_Int32 load_flags )
482 {
483 BDF_Face face = (BDF_Face)FT_SIZE_FACE( size );
484 FT_Error error = BDF_Err_Ok;
485 FT_Bitmap* bitmap = &slot->bitmap;
486 bdf_glyph_t glyph;
487 int bpp = face->bdffont->bpp;
488 int i, j, count;
489 unsigned char *p, *pp;
490
491 FT_Memory memory = face->bdffont->memory;
492
493 FT_UNUSED( load_flags );
494
495
496 if ( !face )
497 {
498 error = BDF_Err_Invalid_Argument;
499 goto Exit;
500 }
501
502 if ( glyph_index > 0 )
503 glyph_index--;
504
505 /* slot, bitmap => freetype, glyph => bdflib */
506 glyph = face->bdffont->glyphs[glyph_index];
507
508 bitmap->rows = glyph.bbx.height;
509 bitmap->width = glyph.bbx.width;
510
511 if ( bpp == 1 )
512 {
513 bitmap->pixel_mode = FT_PIXEL_MODE_MONO;
514 bitmap->pitch = glyph.bpr;
515
516 if ( FT_NEW_ARRAY( bitmap->buffer, glyph.bytes ) )
517 goto Exit;
518 FT_MEM_COPY( bitmap->buffer, glyph.bitmap, glyph.bytes );
519 }
520 else
521 {
522 /* blow up pixmap to have 8 bits per pixel */
523 bitmap->pixel_mode = FT_PIXEL_MODE_GRAY;
524 bitmap->pitch = bitmap->width;
525
526 if ( FT_NEW_ARRAY( bitmap->buffer, bitmap->rows * bitmap->pitch ) )
527 goto Exit;
528
529 switch ( bpp )
530 {
531 case 2:
532 bitmap->num_grays = 4;
533
534 count = 0;
535 p = glyph.bitmap;
536
537 for ( i = 0; i < bitmap->rows; i++ )
538 {
539 pp = p;
540
541 /* get the full bytes */
542 for ( j = 0; j < ( bitmap->width >> 2 ); j++ )
543 {
544 bitmap->buffer[count++] = (FT_Byte)( ( *pp & 0xC0 ) >> 6 );
545 bitmap->buffer[count++] = (FT_Byte)( ( *pp & 0x30 ) >> 4 );
546 bitmap->buffer[count++] = (FT_Byte)( ( *pp & 0x0C ) >> 2 );
547 bitmap->buffer[count++] = (FT_Byte)( *pp & 0x03 );
548
549 pp++;
550 }
551
552 /* get remaining pixels (if any) */
553 switch ( bitmap->width & 3 )
554 {
555 case 3:
556 bitmap->buffer[count++] = (FT_Byte)( ( *pp & 0xC0 ) >> 6 );
557 /* fall through */
558 case 2:
559 bitmap->buffer[count++] = (FT_Byte)( ( *pp & 0x30 ) >> 4 );
560 /* fall through */
561 case 1:
562 bitmap->buffer[count++] = (FT_Byte)( ( *pp & 0x0C ) >> 2 );
563 /* fall through */
564 case 0:
565 break;
566 }
567
568 p += glyph.bpr;
569 }
570 break;
571
572 case 4:
573 bitmap->num_grays = 16;
574
575 count = 0;
576 p = glyph.bitmap;
577
578 for ( i = 0; i < bitmap->rows; i++ )
579 {
580 pp = p;
581
582 /* get the full bytes */
583 for ( j = 0; j < ( bitmap->width >> 1 ); j++ )
584 {
585 bitmap->buffer[count++] = (FT_Byte)( ( *pp & 0xF0 ) >> 4 );
586 bitmap->buffer[count++] = (FT_Byte)( *pp & 0x0F );
587
588 pp++;
589 }
590
591 /* get remaining pixel (if any) */
592 switch ( bitmap->width & 1 )
593 {
594 case 1:
595 bitmap->buffer[count++] = (FT_Byte)( ( *pp & 0xF0 ) >> 4 );
596 /* fall through */
597 case 0:
598 break;
599 }
600
601 p += glyph.bpr;
602 }
603 break;
604
605 case 8:
606 bitmap->num_grays = 256;
607
608 FT_MEM_COPY( bitmap->buffer, glyph.bitmap,
609 bitmap->rows * bitmap->pitch );
610 break;
611 }
612 }
613
614 slot->bitmap_left = 0;
615 slot->bitmap_top = glyph.bbx.ascent;
616
617 /* FZ XXX: TODO: vertical metrics */
618 slot->metrics.horiAdvance = glyph.dwidth << 6;
619 slot->metrics.horiBearingX = glyph.bbx.x_offset << 6;
620 slot->metrics.horiBearingY = ( glyph.bbx.y_offset +
621 glyph.bbx.height ) << 6;
622 slot->metrics.width = bitmap->width << 6;
623 slot->metrics.height = bitmap->rows << 6;
624
625 slot->linearHoriAdvance = (FT_Fixed)glyph.dwidth << 16;
626 slot->format = FT_GLYPH_FORMAT_BITMAP;
627 slot->flags = FT_GLYPH_OWN_BITMAP;
628
629 Exit:
630 return error;
631 }

395 /* FT_ENC_TAG */
396 /* */
397 /* <Description> */
398 /* This macro converts four letter tags into an unsigned long. It is */
399 /* used to define "encoding" identifiers (see @FT_Encoding). */
400 /* */
401 /* <Note> */
402 /* Since many 16bit compilers don't like 32bit enumerations, you */
403 /* should redefine this macro in case of problems to something like */
404 /* this: */
405 /* */
406 /* #define FT_ENC_TAG( value, a, b, c, d ) (value) */
407 /* */
408 /* to get a simple enumeration without assigning special numbers. */
409 /* */
410
411#ifndef FT_ENC_TAG
412#define FT_ENC_TAG( value, a, b, c, d ) \
413 value = ( ( (FT_UInt32)(a) << 24 ) | \
414 ( (FT_UInt32)(b) << 16 ) | \
415 ( (FT_UInt32)(c) << 8 ) | \
416 (FT_UInt32)(d) )
417
418#endif /* FT_ENC_TAG */

422 /* */
423 /* <Enum> */
424 /* FT_Encoding */
425 /* */
426 /* <Description> */
427 /* An enumeration used to specify encodings supported by charmaps. */
428 /* Used in the @FT_Select_Charmap API function. */
429 /* */
430 /* <Note> */
431 /* Because of 32-bit charcodes defined in Unicode (i.e., surrogates), */
432 /* all character codes must be expressed as FT_Longs. */
433 /* */
434 /* The values of this type correspond to specific character */
435 /* repertories (i.e. charsets), and not to text encoding methods */
436 /* (like UTF-8, UTF-16, GB2312_EUC, etc.). */
437 /* */
438 /* Other encodings might be defined in the future. */
439 /* */
440 /* <Values> */
441 /* FT_ENCODING_NONE :: */
442 /* The encoding value 0 is reserved. */
443 /* */
444 /* FT_ENCODING_UNICODE :: */
445 /* Corresponds to the Unicode character set. This value covers */
446 /* all versions of the Unicode repertoire, including ASCII and */
447 /* Latin-1. Most fonts include a Unicode charmap, but not all */
448 /* of them. */
449 /* */
450 /* FT_ENCODING_MS_SYMBOL :: */
451 /* Corresponds to the Microsoft Symbol encoding, used to encode */
452 /* mathematical symbols in the 32..255 character code range. For */
453 /* more information, see `http://www.ceviz.net/symbol.htm'. */
454 /* */
455 /* FT_ENCODING_MS_SJIS :: */
456 /* Corresponds to Microsoft's Japanese SJIS encoding. More info */
457 /* at `http://langsupport.japanreference.com/encoding.shtml'. */
458 /* See note on multi-byte encodings below. */
459 /* */
460 /* FT_ENCODING_MS_GB2312 :: */
461 /* Corresponds to the encoding system for Simplified Chinese, as */
462 /* used in China. Only found in some TrueType fonts. */
463 /* */
464 /* FT_ENCODING_MS_BIG5 :: */
465 /* Corresponds to the encoding system for Traditional Chinese, as */
466 /* used in Taiwan and Hong Kong. Only found in some TrueType fonts. */
467 /* */
468 /* FT_ENCODING_MS_WANSUNG :: */
469 /* Corresponds to the Korean encoding system known as Wansung. */
470 /* This is a Microsoft encoding that is only found in some TrueType */
471 /* fonts. For more information, see */
472 /* `http://www.microsoft.com/typography/unicode/949.txt'. */
473 /* */
474 /* FT_ENCODING_MS_JOHAB :: */
475 /* The Korean standard character set (KS C-5601-1992), which */
476 /* corresponds to Windows code page 1361. This character set */
477 /* includes all possible Hangeul character combinations. */
478 /* Only found on some rare TrueType fonts. */
479 /* */
480 /* FT_ENCODING_ADOBE_LATIN_1 :: */
481 /* Corresponds to a Latin-1 encoding as defined in a Type 1 */
482 /* Postscript font. It is limited to 256 character codes. */
483 /* */
484 /* FT_ENCODING_ADOBE_STANDARD :: */
485 /* Corresponds to the Adobe Standard encoding, as found in Type 1, */
486 /* CFF, and OpenType/CFF fonts. It is limited to 256character */
487 /* codes. */
488 /* */
489 /* FT_ENCODING_ADOBE_EXPERT :: */
490 /* Corresponds to the Adobe Expert encoding, as found in Type 1, */
491 /* CFF, and OpenType/CFF fonts. It is limited to 256 character */
492 /* codes. */
493 /* */
494 /* FT_ENCODING_ADOBE_CUSTOM :: */
495 /* Corresponds to a custom encoding, as found in Type 1, CFF, and */
496 /* OpenType/CFF fonts. It is limited to 256 character codes. */
497 /* */
498 /* FT_ENCODING_APPLE_ROMAN :: */
499 /* Corresponds to the 8-bit Apple roman encoding. Many TrueType and */
500 /* OpenType fonts contain a charmap for this encoding, since older */
501 /* versions of Mac OS are able to use it. */
502 /* */
503 /* FT_ENCODING_OLD_LATIN_2 :: */
504 /* This value is deprecated and was never used nor reported by */
505 /* FreeType. Don't use or test for it. */
506 /* */
507 /* <Note> */
508 /* By default, FreeType automatically synthetizes a Unicode charmap */
509 /* for Postscript fonts, using their glyph names dictionaries. */
510 /* However, it will also report the encodings defined explicitly in */
511 /* the font file, for the cases when they are needed, with the Adobe */
512 /* values as well. */
513 /* */
514 typedef enum FT_Encoding_
515 {
516 FT_ENC_TAG( FT_ENCODING_NONE, 0, 0, 0, 0 ),
517
518 FT_ENC_TAG( FT_ENCODING_MS_SYMBOL, 's', 'y', 'm', 'b' ),
519 FT_ENC_TAG( FT_ENCODING_UNICODE, 'u', 'n', 'i', 'c' ),
520
521 FT_ENC_TAG( FT_ENCODING_MS_SJIS, 's', 'j', 'i', 's' ),
522 FT_ENC_TAG( FT_ENCODING_MS_GB2312, 'g', 'b', ' ', ' ' ),
523 FT_ENC_TAG( FT_ENCODING_MS_BIG5, 'b', 'i', 'g', '5' ),
524 FT_ENC_TAG( FT_ENCODING_MS_WANSUNG, 'w', 'a', 'n', 's' ),
525 FT_ENC_TAG( FT_ENCODING_MS_JOHAB, 'j', 'o', 'h', 'a' ),
526
527 FT_ENC_TAG( FT_ENCODING_ADOBE_STANDARD, 'A', 'D', 'O', 'B' ),
528 FT_ENC_TAG( FT_ENCODING_ADOBE_EXPERT, 'A', 'D', 'B', 'E' ),
529 FT_ENC_TAG( FT_ENCODING_ADOBE_CUSTOM, 'A', 'D', 'B', 'C' ),

2005 /****************************************************************************
2006 *
2007 * @enum:
2008 * FT_LOAD_XXX
2009 *
2010 * @description:
2011 * A list of bit-field constants, used with @FT_Load_Glyph to indicate
2012 * what kind of operations to perform during glyph loading.
2013 *
2014 * @values:
2015 * FT_LOAD_DEFAULT ::
2016 * Corresponding to 0, this value is used a default glyph load. In this
2017 * case, the following will happen:
2018 *
2019 * 1. FreeType looks for a bitmap for the glyph corresponding to the
2020 * face's current size. If one is found, the function returns. The
2021 * bitmap data can be accessed from the glyph slot (see note below).
2022 *
2023 * 2. If no embedded bitmap is searched or found, FreeType looks for a
2024 * scalable outline. If one is found, it is loaded from the font
2025 * file, scaled to device pixels, then "hinted" to the pixel grid in
2026 * order to optimize it. The outline data can be accessed from the
2027 * glyph slot (see note below).
2028 *
2029 * Note that by default, the glyph loader doesn't render outlines into
2030 * bitmaps. The following flags are used to modify this default
2031 * behaviour to more specific and useful cases.
2032 *
2033 * FT_LOAD_NO_SCALE ::
2034 * Don't scale the vector outline being loaded to 26.6 fractional
2035 * pixels, but kept in font units. Note that this also disables
2036 * hinting and the loading of embedded bitmaps. You should only use it
2037 * when you want to retrieve the original glyph outlines in font units.
2038 *
2039 * FT_LOAD_NO_HINTING ::
2040 * Don't hint glyph outlines after their scaling to device pixels.
2041 * This generally generates "blurrier" glyphs in anti-aliased modes.
2042 *
2043 * This flag is ignored if @FT_LOAD_NO_SCALE is set.
2044 *
2045 * FT_LOAD_RENDER ::
2046 * Render the glyph outline immediately into a bitmap before the glyph
2047 * loader returns. By default, the glyph is rendered for the
2048 * @FT_RENDER_MODE_NORMAL mode, which corresponds to 8-bit anti-aliased
2049 * bitmaps using 256 opacity levels. You can use either
2050 * @FT_LOAD_TARGET_MONO or @FT_LOAD_MONOCHROME to render 1-bit
2051 * monochrome bitmaps.
2052 *
2053 * This flag is ignored if @FT_LOAD_NO_SCALE is set.
2054 *
2055 * FT_LOAD_NO_BITMAP ::
2056 * Don't look for bitmaps when loading the glyph. Only scalable
2057 * outlines will be loaded when available, and scaled, hinted, or
2058 * rendered depending on other bit flags.
2059 *
2060 * This does not prevent you from rendering outlines to bitmaps
2061 * with @FT_LOAD_RENDER, however.
2062 *
2063 * FT_LOAD_VERTICAL_LAYOUT ::
2064 * Prepare the glyph image for vertical text layout. This basically
2065 * means that `face.glyph.advance' will correspond to the vertical
2066 * advance height (instead of the default horizontal advance width),
2067 * and that the glyph image will be translated to match the vertical
2068 * bearings positions.
2069 *
2070 * FT_LOAD_FORCE_AUTOHINT ::
2071 * Force the use of the FreeType auto-hinter when a glyph outline is
2072 * loaded. You shouldn't need this in a typical application, since it
2073 * is mostly used to experiment with its algorithm.
2074 *
2075 * FT_LOAD_CROP_BITMAP ::
2076 * Indicates that the glyph loader should try to crop the bitmap (i.e.,
2077 * remove all space around its black bits) when loading it. This is
2078 * only useful when loading embedded bitmaps in certain fonts, since
2079 * bitmaps rendered with @FT_LOAD_RENDER are always cropped by default.
2080 *
